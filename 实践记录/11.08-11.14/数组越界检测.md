  void wrapped_memcpy()
  {
      char data[50];
      memset(data, 'A', 50-1); /* fill with 'A's */
      data[50-1] = '\0'; /* null terminate */
  	
  
  	char dest[100];
  	memset(dest, 'C', 100-1);
  	dest[100-1] = '\0'; /* null terminate */
  	
  	{      
  	    memcpy(dest, data, 100);
  		//memcpy(dest, data, strlen(dest)*sizeof(char));
  	    dest[100-1] = '\0';
  	}
  
  }
  
  int main(int argc, char * argv[])
  {
      wrapped_memcpy();
      return 0;
  }
